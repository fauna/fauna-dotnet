110 ISerializer<T> serializer,
111 MappingContext ctx,
112 QueryOptions? queryOptions,
113 CancellationToken cancel)
114 {
115 if (query == null)
116 {
117 throw new ArgumentNullException(nameof(query));
118 }
119
120 var finalOptions = QueryOptions.GetFinalQueryOptions(_config.DefaultQueryOptions, queryOptions);
121 var headers = GetRequestHeaders(finalOptions);
122
123 using var stream = new MemoryStream();
124 Serialize(stream, query, ctx);
125
126 using var httpResponse = await _connection.DoPostAsync(
127 QueryUriPath,
128 stream,
129 headers,
130 GetRequestTimeoutWithBuffer(finalOptions.QueryTimeout),
131 cancel);
132 var body = await httpResponse.Content.ReadAsStringAsync(cancel);
133 var res = QueryResponse.GetFromResponseBody<T>(ctx, serializer, httpResponse.StatusCode, body);
134 switch (res)

136 case QuerySuccess<T> success:

138 StatsCollector?.Add(res.Stats);
139 return success;
140 case QueryFailure failure:
141 StatsCollector?.Add(res.Stats);
142 throw ExceptionHandler.FromQueryFailure(ctx, failure);
143 default:
144 throw ExceptionHandler.FromRawResponse(body, httpResponse);
145 }
