94 public bool Read()
95 {
96 _taggedTokenValue = null;
97
98 if (_bufferedTokenType != null)
99 {
100 CurrentTokenType = (TokenType)_bufferedTokenType;
101 _bufferedTokenType = null;
102 if (_closers.Contains(CurrentTokenType))
103 {
104 _tokenStack.Pop();
105 }
106 return true;
107 }
108
109 if (!Advance())
110 {
111 return false;
112 }
113
114 switch (_json.TokenType)
115 {
116 case JsonTokenType.PropertyName:
117 CurrentTokenType = TokenType.FieldName;
118 break;
119 case JsonTokenType.None:
120 break;
121 case JsonTokenType.StartObject:
122 HandleStartObject();
123 break;
124 case JsonTokenType.EndObject:
125 HandleEndObject();
126 break;
127 case JsonTokenType.StartArray:
128 _tokenStack.Push(TokenType.StartArray);
129 CurrentTokenType = TokenType.StartArray;
130 break;
131 case JsonTokenType.EndArray:
132 _tokenStack.Pop();
133 CurrentTokenType = TokenType.EndArray;
134 break;
135 case JsonTokenType.String:
136 CurrentTokenType = TokenType.String;
137 break;
138 case JsonTokenType.True:
139 CurrentTokenType = TokenType.True;
140 break;
141 case JsonTokenType.False:
142 CurrentTokenType = TokenType.False;
143 break;
144 case JsonTokenType.Null:
145 CurrentTokenType = TokenType.Null;
146 break;
147 case JsonTokenType.Comment:
148 case JsonTokenType.Number:
149 default:
150 throw new SerializationException($"Unhandled JSON token type {_json.TokenType}.");
151 }
152
153 return true;
154 }

537 private void HandleStartObject()
538 {
539 AdvanceTrue();
540
541 switch (_json.TokenType)
542 {
543 case JsonTokenType.PropertyName:
544 switch (_json.GetString())
545 {
546 case "@date":
547 HandleTaggedString(TokenType.Date);
548 break;
549 case "@doc":
550 AdvanceTrue();
551 CurrentTokenType = TokenType.StartDocument;
552 _tokenStack.Push(TokenType.StartDocument);
553 break;
554 case "@double":
555 HandleTaggedString(TokenType.Double);
556 break;
557 case "@int":
558 HandleTaggedString(TokenType.Int);
559 break;
560 case "@long":
561 HandleTaggedString(TokenType.Long);
562 break;
563 case "@mod":
564 HandleTaggedString(TokenType.Module);
565 break;
566 case "@stream":
567 HandleTaggedString(TokenType.Stream);
568 break;
569 case "@object":
570 AdvanceTrue();
571 CurrentTokenType = TokenType.StartObject;
572 _tokenStack.Push(TokenTypeInternal.StartEscapedObject);
573 break;
574 case "@ref":
575 AdvanceTrue();
576 CurrentTokenType = TokenType.StartRef;
577 _tokenStack.Push(TokenType.StartRef);
578 break;
579 case "@set":
580 AdvanceTrue();
581 CurrentTokenType = TokenType.StartPage;
582 _tokenStack.Push(TokenType.StartPage);
583 break;
584 case "@time":
585 HandleTaggedString(TokenType.Time);
586 break;
587 default:
588 _bufferedTokenType = TokenType.FieldName;
589 _tokenStack.Push(TokenType.StartObject);
590 CurrentTokenType = TokenType.StartObject;
591 break;
592 }
593 break;
594 case JsonTokenType.EndObject:
595 _bufferedTokenType = TokenType.EndObject;
596 _tokenStack.Push(TokenType.StartObject);
597 CurrentTokenType = TokenType.StartObject;
598 break;
599 default:
600 throw new SerializationException($"Unexpected token following StartObject: {_json.TokenType}");
601 }
602 }
603
604 private void HandleEndObject()
605 {
606 var startToken = _tokenStack.Pop();
607 switch (startToken)
608 {
609 case TokenType.StartDocument:
610 CurrentTokenType = TokenType.EndDocument;
611 AdvanceTrue();
612 break;
613 case TokenType.StartPage:
614 CurrentTokenType = TokenType.EndPage;
615 AdvanceTrue();
616 break;
617 case TokenType.StartRef:
618 CurrentTokenType = TokenType.EndRef;
619 AdvanceTrue();
620 break;
621 case TokenTypeInternal.StartEscapedObject:
622 CurrentTokenType = TokenType.EndObject;
623 AdvanceTrue();
624 break;
625 case TokenType.StartObject:
626 CurrentTokenType = TokenType.EndObject;
627 break;
628 default:
629 throw new SerializationException($"Unexpected token {startToken}. This might be a bug.");
630 }
631 }
